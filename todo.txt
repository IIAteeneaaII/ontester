import math
from datetime import datetime

def _is_bad_info(v) -> bool:
    """True si el valor es 'basura' y NO debe sobrescribir en DB."""
    if v is None:
        return True
    # float nan
    if isinstance(v, float) and math.isnan(v):
        return True
    s = str(v).strip()
    if s == "":
        return True
    s_up = s.upper()
    return s_up in {"NAN", "NONE", "NULL", "N/A", "NA", "--", "---", "SIN_DATO"}

def _day_yyyy_mm_dd(s: str) -> str:
    """Toma 'YYYY-MM-DD' de un ISO (o de now)."""
    return (s or "")[:10]

def actualizar_operacion(payload, modo, id_user):
    info  = payload.get("info", {}) or {}
    tests = payload.get("tests", {}) or {}

    tipo = (modo or "").strip().upper()
    if tipo == "RETESTEO":
        tipo = "RETEST"
    if tipo not in {"ETIQUETA", "TESTEO", "RETEST"}:
        return -1

    # version actual del tester (FK)
    versRow = extraer_ultimo("catalog_meta")
    if not versRow:
        raise ValueError("No hay catalog_meta inicializado")
    id_catalog_meta = versRow["id"]

    # station del usuario (última relación)
    ultimoStationUser = get_ultimo_user_station_por_usuario(id_user)
    if not ultimoStationUser:
        return -1
    id_station = ultimoStationUser["id_station"]

    # settings vigente (si así lo manejas)
    ultima_setting = extraer_ultimo("settings")
    if not ultima_setting:
        return -1
    id_settings = ultima_setting["id"]

    from src.backend.endpoints.conexion import norm_result, norm_power

    false_means = "FAIL"

    fecha_test = info.get("fecha_test")
    # para ubicar el registro del día (si no viene fecha_test, usamos hoy local)
    hoy = _day_yyyy_mm_dd(fecha_test) or datetime.now().astimezone().date().isoformat()

    # Campos info: si vienen “mal”, NO sobrescriben (mandamos None)
    def clean_info(key: str):
        v = info.get(key)
        return None if _is_bad_info(v) else v

    params = {
        "id_station": id_station,
        "id_user": id_user,
        "id_settings": id_settings,
        "id_catalog_meta": id_catalog_meta,
        "tipo": tipo,

        # para localizar el registro del día
        "day": hoy,

        # info (con protección)
        "fecha_test": None if _is_bad_info(fecha_test) else fecha_test,
        "modelo":   clean_info("modelo"),
        "sn":       clean_info("sn"),
        "mac":      clean_info("mac"),
        "sftVer":   clean_info("sftVer"),
        "wifi24":   clean_info("wifi24"),
        "wifi5":    clean_info("wifi5"),
        "passWifi": clean_info("passWifi"),

        # tests (estos sí se actualizan siempre)
        "ping":  norm_result(tests.get("ping"),  false_means=false_means),
        "reset": norm_result(tests.get("reset"), false_means=false_means),
        "usb":   norm_result(tests.get("usb"),   false_means=false_means),
        "tx":    norm_power(tests.get("tx"), "tx"),
        "rx":    norm_power(tests.get("rx"), "rx"),
        "w24":   norm_result(tests.get("w24"), false_means=false_means),
        "w5":    norm_result(tests.get("w5"),  false_means=false_means),
        "sftU":  norm_result(tests.get("sftU"), false_means=false_means),

        "valido": int(bool(payload.get("valido"))),
    }

    sql = """
    UPDATE operations
    SET
        id_station      = :id_station,
        id_user         = :id_user,
        id_settings     = :id_settings,
        id_catalog_meta = :id_catalog_meta,
        tipo            = :tipo,

        -- info: NO sobrescribir si viene "malo"
        fecha_test = COALESCE(:fecha_test, fecha_test),
        modelo     = COALESCE(:modelo, modelo),
        sn         = COALESCE(:sn, sn),
        mac        = COALESCE(:mac, mac),
        sftVer     = COALESCE(:sftVer, sftVer),
        wifi24     = COALESCE(:wifi24, wifi24),
        wifi5      = COALESCE(:wifi5, wifi5),
        passWifi   = COALESCE(:passWifi, passWifi),

        -- tests: siempre actualizar
        ping  = :ping,
        reset = :reset,
        usb   = :usb,
        tx    = :tx,
        rx    = :rx,
        w24   = :w24,
        w5    = :w5,
        sftU  = :sftU,

        valido = :valido
    WHERE id = (
        SELECT id
        FROM operations
        WHERE id_user = :id_user
          AND tipo = :tipo
          AND substr(fecha_test, 1, 10) = :day
        ORDER BY fecha_test DESC, id DESC
        LIMIT 1
    );
    """

    with get_conn() as con:
        cur = con.execute(sql, params)
        con.commit()
        return cur.rowcount  # 1 si actualizó, 0 si no encontró registro del día
